- 输出杨辉三角形前 `n` 行（`1 <= n <= 30`）。
  要求实现**一维数组原地更新**版（从右向左更新，避免数据污染）：
- ```C
  #include <stdio.h>
  void pp(int n);
  
  int main(){
      int n;
      scanf("%d",&n);
      pp(n);
  }
  /* 避免数据污染的原理：
  假设当前数组是：[1, 2, 1]（第三行）
  我们要计算第四行：
  
  从左往右（错误）：
  j=1: line[1] = line[1] + line[0] = 2 + 1 = 3 → [1, 3, 1]
  j=2: line[2] = line[2] + line[1] = 1 + 3 = 4 → [1, 3, 4] ❌ 错了！（使用line[2]导致原数据被覆盖）
  
  从右往左（正确）：
  j=2: line[2] = line[2] + line[1] = 1 + 2 = 3 → [1, 2, 3]
  j=1: line[1] = line[1] + line[0] = 2 + 1 = 3 → [1, 3, 3] ✅ 正确！*/
  void pp(int n){
      int a[31] = {0};                    #题目中n<=30，初始化了全为零的数组
      a[0] = 1;                           #初始
      for (int i = 0; i < n; i++){        #循环是为了输出n次
          for (int j = i; j > 0; j--){    #从右往左更新数据，简单地说就是先移动再使用
              a[j] += a[j-1];
          }
          for (int j = 0; j <= i; j++){
              printf("%d", a[j]);
              if(j < i){
                  printf(" ");            #控制空格的输出，每行最后一个不用空格
              }
          }
          printf("\n");
      }
  }
  ```
- 程设课上的一行输出版本：
- ```C
  #include <stdio.h>
  #include <string.h>
  void pascal(int n) {
      int t[n+1];
      // Initialize t[] to 0
      memset(t, 0, sizeof(t));             #这一行与上一个代码块的方法不同！
      // Initialize t[i] to 1 for all i
      for (int i = 0; i <= n; i++) t[i] = 1;   
      // Update other entries
      for (int i = 1; i < n; i++)          #这行是i=1，上一个代码块是i=0 
          for (int j = i; j >= 1; j--)
              t[j] = t[j-1] + t[j];
      // Print Pascal Triangle
      for (int j = 0; j <= n; j++) printf("%d ", t[j]);
      printf("\n");
  }
  int main() {
      int n;
      scanf("%d", &n);
      pascal(n);
      return 0;
  }
